Basic elements and syntax
Console examples: 

>>> x = 5
>>> y = 1.5
>>> x*y
7.5
>>> x=("1","a","2")
>>> x
('1', 'a', '2')
>>> x=("this", "is", "a", "tuple")
>>> x
('this', 'is', 'a', 'tuple')
>>> y=["this", "is", "a", "list"]
>>> y
['this', 'is', 'a', 'list']
>>> type(x)
<type 'tuple'>
>>> type(y)
<type 'list'>

>>> list = ['1', '2', '3', '4', '1', '1','3','5']
>>> set(list)
set(['1', '3', '2', '5', '4'])
>>> zed = "lowercase string"
>>> zed.upper()
'LOWERCASE STRING'
>>> shopping_list = ["eggs","milk","bread"]
>>> shopping_list.append("butter");
>>> shopping_list
['eggs', 'milk', 'bread', 'butter']
>>> a = 1
>>> b = 2
>>> a > b
False
>>> b > a
True
>>> [1,2,3] > [1,2]
True
>>> [1,2] > [1,2,3]
False
>>> bool('foo')
True
>>> print 'foo' if False else 'bar'
bar
